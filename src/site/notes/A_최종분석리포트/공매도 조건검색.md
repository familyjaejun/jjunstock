# 공매도 조건검색 (통합: 5필터 AND + 숏커버 BEST10 + 신용 BEST10)

```dataviewjs
(async () => {
  const CFG = {
    loanFloatPctMin: 10.0,
    contRepayMin: 2,

    enableShortAmt: true,
    shortAmtEokMin: 10,

    enableRecent3Repay: false,
    recent3NetRepayMax: -100000,

    enableTurnover5d: false,
    turnover5dAvgEokMin: 50,

    bestShortN: 10,
    bestCreditN: 10,

    DEBUG: false, // true로 하면 하단에 디버그/에러 출력
  };

  const STOCK_TAG = "증거금/20프로";
  const PATH_CONTAINS = ["증거금20/종목/", "증거금20/증거금20/종목/"];

  // ── 유틸 ─────────────────────────────────
  const toStr = (x) => String(x ?? "");
  const strip = (s) => toStr(s).trim();
  const rmComma = (s) => strip(s).replace(/,/g, "");
  const isDash = (s) => /^[-–—]+$/.test(strip(s).replace(/\s/g, ""));

  function toInt(s) {
    s = strip(s);
    if (!s || isDash(s)) return null;
    const m = rmComma(s).match(/-?\d+/);
    return m ? parseInt(m[0], 10) : null;
  }
  function toFloat(s) {
    s = strip(s);
    if (!s || isDash(s)) return null;
    s = rmComma(s).replace(/%/g, "");
    const m = s.match(/-?\d+(?:\.\d+)?/);
    return m ? parseFloat(m[0]) : null;
  }
  function parseSignedInt(s) {
    s = strip(s);
    if (!s || isDash(s)) return null;
    let sign = 1;
    if (s.includes("▼")) sign = -1;
    if (s.includes("▲")) sign = 1;
    s = s.replace(/[▲▼]/g, "");
    const n = toInt(s);
    return n === null ? null : sign * n;
  }
  const eok = (krw) => (krw === null ? null : krw / 100000000);

  const fmtPct = (x, d=2) => (typeof x === "number" && !Number.isNaN(x)) ? `${x.toFixed(d)}%` : "-";
  const fmtEok = (x, d=2) => (typeof x === "number" && !Number.isNaN(x)) ? x.toFixed(d) : "-";

  function dedupPages(arr) {
    const m = new Map();
    for (const p of arr) m.set(p.file.path, p);
    return Array.from(m.values());
  }

  function getAllPages() {
    const all = dv.pages();

    const byPath = all.where(p => {
      const path = toStr(p.file?.path);
      return PATH_CONTAINS.some(key => path.includes(key));
    }).array();

    const byTag = all.where(p => {
      const t1 = Array.isArray(p.tags) ? p.tags : [];
      const t2 = Array.isArray(p.file?.tags) ? p.file.tags : [];
      const tags = [...t1, ...t2].map(x => toStr(x).replace(/^#/, ""));
      return tags.includes(STOCK_TAG);
    }).array();

    return { byPath, byTag, pages: dedupPages([...byPath, ...byTag]) };
  }

  // ── 본문 파싱 ─────────────────────────────
  const RE_CONT_REPAY = /연속\s*상환\s*:\s*\*{0,8}\s*([0-9]+)\s*일/;
  const RE_FLOAT_SHARES  = /유통주식\s*\(유통비율\)\s*\|\s*([0-9,]+)\s*주/;
  const RE_LISTED_SHARES = /상장주식수\s*\|\s*([0-9,]+)\s*주/;

  function normalizeLine(line) {
    line = strip(line);
    while (line.startsWith(">")) line = line.replace(/^>\s*/, "");
    return strip(line);
  }
  function splitCells(line) {
    return normalizeLine(line).split("|").map(s => strip(s)).filter(Boolean);
  }
  const isDateCell = (x) => /^\d{2}\/\d{2}\/\d{2}$/.test(strip(x));

  function idxOf(header, includesAny, excludesAny=[]) {
    const h = (header ?? []).map(x => strip(x));
    for (let i = 0; i < h.length; i++) {
      const t = h[i];
      const ok = includesAny.some(k => t.includes(k));
      const bad = excludesAny.some(k => t.includes(k));
      if (ok && !bad) return i;
    }
    return -1;
  }

  function parseTableAt(lines, startIdx, headerTest, maxRows=30, maxScan=120) {
    for (let i = startIdx; i < Math.min(lines.length, startIdx + maxScan); i++) {
      const ln = normalizeLine(lines[i]);
      if (!ln.startsWith("|")) continue;
      if (!headerTest(ln)) continue;

      const header = splitCells(ln);
      const rows = [];

      for (let j = i + 2; j < lines.length && rows.length < maxRows; j++) {
        const rln = normalizeLine(lines[j]);
        if (!rln.startsWith("|")) break;

        const cells = splitCells(rln);
        if (!cells.length) continue;
        if (!isDateCell(cells[0])) break;

        rows.push(cells);
      }
      return { header, rows, headerLine: ln };
    }
    return null;
  }

  function findFirstAnchor(lines, keywords) {
    for (let i = 0; i < lines.length; i++) {
      const ln = normalizeLine(lines[i]);
      if (keywords.some(k => ln.includes(k))) return i;
    }
    return -1;
  }

  // 헤더 테스트(더 관대)
  const isLoanHeader = (ln) => {
    const s = normalizeLine(ln);
    if (!s.includes("날짜")) return false;
    const hasSigned = s.includes("대차체결") || (s.includes("대차") && s.includes("체결"));
    const hasRepaid = s.includes("대차상환") || (s.includes("대차") && s.includes("상환"));
    const hasBal = s.includes("대차잔고") || s.includes("잔고주수") || (s.includes("대차") && s.includes("잔고"));
    const hasDelta = s.includes("대차증감") || s.includes("증감");
    return hasSigned && hasRepaid && hasBal && hasDelta;
  };

  const isShortHeader = (ln) => {
    const s = normalizeLine(ln);
    if (!s.includes("날짜")) return false;
    const hasQty = s.includes("공매도량") || s.includes("공매도수량");
    const hasPx  = s.includes("평균가");
    const hasVol = s.includes("거래량");
    return hasQty && hasPx && hasVol;
  };

  const isCreditHeader = (ln) => {
    const s = normalizeLine(ln);
    if (!s.includes("날짜")) return false;
    const hasNew = s.includes("신용신규") || s.includes("신규");
    const hasRp  = s.includes("신용상환") || (s.includes("신용") && s.includes("상환"));
    const hasBal = s.includes("신용잔고") || (s.includes("신용") && s.includes("잔고"));
    const hasSup = s.includes("공여율");
    return hasBal && hasRp && hasSup && hasNew;
  };

  function parseOne(text) {
    text = toStr(text);
    const lines = text.split("\n");

    const mCont = text.match(RE_CONT_REPAY);
    const contRepay = mCont ? parseInt(mCont[1], 10) : null;

    const mFloat = text.match(RE_FLOAT_SHARES);
    const floatShares = mFloat ? toInt(mFloat[1]) : null;

    const mListed = text.match(RE_LISTED_SHARES);
    const listedShares = mListed ? toInt(mListed[1]) : null;

    // ── 대차
    let loanTbl = parseTableAt(lines, 0, isLoanHeader, 20, lines.length);
    if (!loanTbl) {
      const a = findFirstAnchor(lines, ["대차거래 상세", "▸ 대차거래 상세"]);
      if (a >= 0) loanTbl = parseTableAt(lines, a, isLoanHeader, 20, 120);
    }

    let loanBalLatest = null;
    let loanDelta3 = null;

    if (loanTbl && loanTbl.rows.length) {
      const h = loanTbl.header;
      const iDelta = idxOf(h, ["대차증감","증감"]);
      const iBal   = idxOf(h, ["대차잔고","잔고주수","잔고"], ["잔고율"]);

      const latest = loanTbl.rows[0];
      loanBalLatest = (iBal >= 0 && latest[iBal] !== undefined) ? toInt(latest[iBal]) : null;

      if (iDelta >= 0) {
        const last3 = loanTbl.rows.slice(0, 3);
        const vals = last3
          .map(r => (r[iDelta] !== undefined ? parseSignedInt(r[iDelta]) : null))
          .filter(v => v !== null);
        if (vals.length) loanDelta3 = vals.reduce((a,b)=>a+b,0);
      }
    }

    let loanPctTotal = null, loanPctFloat = null;
    if (loanBalLatest !== null && listedShares !== null && listedShares > 0) loanPctTotal = (loanBalLatest / listedShares) * 100;
    if (loanBalLatest !== null && floatShares  !== null && floatShares  > 0) loanPctFloat = (loanBalLatest / floatShares)  * 100;
    const hasLoanData = loanBalLatest !== null && loanBalLatest > 0;

    // ── 공매도
    let shortTbl = parseTableAt(lines, 0, isShortHeader, 20, lines.length);
    if (!shortTbl) {
      const a = findFirstAnchor(lines, ["시세·공매도 상세", "▸ 시세·공매도 상세"]);
      if (a >= 0) shortTbl = parseTableAt(lines, a, isShortHeader, 20, 120);
    }

    let shortAmtEok = null, turnover5dAvgEok = null;
    if (shortTbl && shortTbl.rows.length) {
      const h = shortTbl.header;
      const iQty = idxOf(h, ["공매도량","공매도수량"]);
      const iPx  = idxOf(h, ["평균가"]);
      const iVol = idxOf(h, ["거래량"]);

      const latest = shortTbl.rows[0];
      const qty = (iQty >= 0 && latest[iQty] !== undefined) ? toInt(latest[iQty]) : null;
      const px  = (iPx  >= 0 && latest[iPx]  !== undefined) ? toInt(latest[iPx])  : null;
      if (qty !== null && px !== null) shortAmtEok = eok(qty * px);

      const last5 = shortTbl.rows.slice(0, 5);
      const amts = [];
      for (const r of last5) {
        const v = (iVol >= 0 && r[iVol] !== undefined) ? toInt(r[iVol]) : null;
        const p = (iPx  >= 0 && r[iPx]  !== undefined) ? toInt(r[iPx])  : null;
        if (v !== null && p !== null) amts.push(eok(v * p));
      }
      if (amts.length) turnover5dAvgEok = amts.reduce((a,b)=>a+b,0)/amts.length;
    }

    // ── 신용
    let creditTbl = parseTableAt(lines, 0, isCreditHeader, 20, lines.length);
    if (!creditTbl) {
      const a = findFirstAnchor(lines, ["신용잔고 상세", "▸ 신용잔고 상세"]);
      if (a >= 0) creditTbl = parseTableAt(lines, a, isCreditHeader, 20, 120);
    }

    let creditRepayLatest = null, creditBalLatest = null, creditSupplyPct = null, creditDelta3 = null, creditPctTotal=null, creditPctFloat=null;
    if (creditTbl && creditTbl.rows.length) {
      const h = creditTbl.header;
      const iNew = idxOf(h, ["신용신규","신규"]);
      const iRp  = idxOf(h, ["신용상환","상환"]);
      const iBal = idxOf(h, ["신용잔고","잔고"], ["잔고율"]);
      const iSup = idxOf(h, ["공여율"]);

      const latest = creditTbl.rows[0];
      const rr = (iRp >=0 && latest[iRp ]!==undefined) ? toInt(latest[iRp ]) : null;
      const bb = (iBal>=0 && latest[iBal]!==undefined) ? toInt(latest[iBal]) : null;
      const sp = (iSup>=0 && latest[iSup]!==undefined) ? toFloat(latest[iSup]) : null;

      creditRepayLatest = rr;
      creditBalLatest = bb;
      creditSupplyPct = sp;

      // 3일 신용증감(주) = (신규 - 상환) 3일 합  ※ 값/로직 유지
      const last3 = creditTbl.rows.slice(0, 3);
      const vals = [];
      for (const r of last3) {
        const n = (iNew>=0 && r[iNew]!==undefined) ? toInt(r[iNew]) : null;
        const x = (iRp >=0 && r[iRp ]!==undefined) ? toInt(r[iRp ]) : null;
        if (n !== null && x !== null) vals.push(n - x);
      }
      if (vals.length) creditDelta3 = vals.reduce((a,b)=>a+b,0);

      if (creditBalLatest !== null && listedShares !== null && listedShares > 0) creditPctTotal = (creditBalLatest / listedShares) * 100;
      if (creditBalLatest !== null && floatShares  !== null && floatShares  > 0) creditPctFloat = (creditBalLatest / floatShares)  * 100;
    }

    const hasCreditData = creditBalLatest !== null && creditBalLatest > 0;

    return {
      contRepay, floatShares, listedShares,

      loanBalLatest, loanPctTotal, loanPctFloat, loanDelta3, hasLoanData,
      shortAmtEok, turnover5dAvgEok,

      creditBalLatest, creditPctTotal, creditPctFloat, creditSupplyPct, creditDelta3, creditRepayLatest, hasCreditData,
    };
  }

  // ── 스캔 ─────────────────────────────────
  const { byPath, byTag, pages } = getAllPages();

  if (!pages.length) {
    dv.header(2, "❌ 스캔 대상 0개");
    dv.paragraph(`- 경로 힌트: ${PATH_CONTAINS.map(x => `\`${x}\``).join(", ")}`);
    dv.paragraph(`- 태그 힌트: \`#${STOCK_TAG}\``);
    return;
  }

  const rows = [];
  const errors = [];
  let loanDataYes = 0, creditDataYes = 0;

  for (const p of pages) {
    let text = "";
    try {
      text = await dv.io.load(p.file.path);
    } catch (e) {
      errors.push({ link: p.file.link, where: "load", msg: String(e?.message ?? e) });
      continue;
    }

    try {
      const d = parseOne(text);
      if (d.hasLoanData) loanDataYes += 1;
      if (d.hasCreditData) creditDataYes += 1;
      rows.push({ link: p.file.link, ...d });
    } catch (e) {
      errors.push({ link: p.file.link, where: "parse", msg: String(e?.message ?? e) });
    }
  }

  // ── ALL PASS ─────────────────────────────
  function passAll(r) {
    if (r.loanPctFloat === null || r.loanPctFloat < CFG.loanFloatPctMin) return false;
    if (r.contRepay === null || r.contRepay < CFG.contRepayMin) return false;

    if (CFG.enableShortAmt) {
      if (r.shortAmtEok === null || r.shortAmtEok < CFG.shortAmtEokMin) return false;
    }
    if (CFG.enableRecent3Repay) {
      // 3일 대차증감(주) <= (음수 기준)
      if (r.loanDelta3 === null || r.loanDelta3 > CFG.recent3NetRepayMax) return false;
    }
    if (CFG.enableTurnover5d) {
      if (r.turnover5dAvgEok === null || r.turnover5dAvgEok < CFG.turnover5dAvgEokMin) return false;
    }
    return true;
  }

  const allPass = rows.filter(passAll).sort((a,b) =>
    (b.loanPctFloat ?? -1) - (a.loanPctFloat ?? -1) ||
    (b.contRepay ?? -1) - (a.contRepay ?? -1) ||
    (b.shortAmtEok ?? -1) - (a.shortAmtEok ?? -1)
  );

  // ── 점수 유틸 ─────────────────────────────
  function minMax(vals) {
    const v = vals.filter(x => typeof x === "number" && !Number.isNaN(x));
    if (!v.length) return { min: 0, max: 0 };
    return { min: Math.min(...v), max: Math.max(...v) };
  }
  function norm(v, mm) {
    if (typeof v !== "number" || Number.isNaN(v)) return 0;
    const den = (mm.max - mm.min);
    if (den <= 0) return 0;
    return (v - mm.min) / den;
  }

  // ── 숏커버 BEST ───────────────────────────
  const mmLoanF  = minMax(rows.map(r => r.loanPctFloat));
  const mmCont   = minMax(rows.map(r => r.contRepay));
  const mmShort  = minMax(rows.map(r => r.shortAmtEok));
  const mmRepay  = minMax(rows.map(r => (typeof r.loanDelta3 === "number" ? Math.max(0, -r.loanDelta3) : null)));
  const mmTurn   = minMax(rows.map(r => r.turnover5dAvgEok));

  let wShort = [
    ["loan", 0.45],
    ["cont", 0.30],
    ["short", CFG.enableShortAmt ? 0.20 : 0.00],
    ["repay", CFG.enableRecent3Repay ? 0.03 : 0.00],
    ["turn",  CFG.enableTurnover5d ? 0.02 : 0.00],
  ];
  const wShortSum = wShort.reduce((a, x) => a + x[1], 0) || 1;
  wShort = wShort.map(([k, v]) => [k, v / wShortSum]);

  function shortScore(r) {
    const repayAbs = (typeof r.loanDelta3 === "number") ? Math.max(0, -r.loanDelta3) : null;
    const parts = {
      loan: norm(r.loanPctFloat, mmLoanF),
      cont: norm(r.contRepay, mmCont),
      short: norm(r.shortAmtEok, mmShort),
      repay: norm(repayAbs, mmRepay),
      turn: norm(r.turnover5dAvgEok, mmTurn),
    };
    const s = wShort.reduce((acc, [k, wt]) => acc + wt * (parts[k] ?? 0), 0);
    return Math.round(s * 1000) / 10;
  }

  const bestShort = rows
    .filter(r => r.loanPctFloat !== null && r.contRepay !== null)
    .map(r => ({ ...r, score: shortScore(r) }))
    .sort((a, b) => b.score - a.score)
    .slice(0, CFG.bestShortN);

  // ── 신용 BEST ─────────────────────────────
  const mmCrF  = minMax(rows.map(r => r.creditPctFloat));
  const mmCrRp = minMax(rows.map(r => (typeof r.creditDelta3 === "number" ? Math.max(0, -r.creditDelta3) : null)));

  function creditScore(r) {
    const repayAbs = (typeof r.creditDelta3 === "number") ? Math.max(0, -r.creditDelta3) : null;
    const parts = {
      cr: norm(r.creditPctFloat, mmCrF),
      rp: norm(repayAbs, mmCrRp),
      turn: norm(r.turnover5dAvgEok, mmTurn),
    };
    const s = (0.60 * parts.cr) + (0.30 * parts.rp) + (0.10 * parts.turn);
    return Math.round(s * 1000) / 10;
  }

  const bestCredit = rows
    .filter(r => r.creditPctFloat !== null)
    .map(r => ({ ...r, score: creditScore(r) }))
    .sort((a, b) => b.score - a.score)
    .slice(0, CFG.bestCreditN);

  // ── 출력 ─────────────────────────────────
  dv.header(2, "공매도 조건검색 (통합)");
  dv.paragraph(`- 스캔대상: 경로(${byPath.length}) + 태그(#${STOCK_TAG}, ${byTag.length})`);
  dv.paragraph(`- 총 노트: **${rows.length}**`);
  dv.paragraph(`- 대차데이터 유효(대차잔고>0): **${loanDataYes}** / ${rows.length}`);
  dv.paragraph(`- 신용데이터 유효(신용잔고>0): **${creditDataYes}** / ${rows.length}`);

  dv.paragraph(`- 용어: **3일 대차증감(주)**=대차증감 3일 합 / **3일 신용증감(주)**=(신규-상환) 3일 합 (음수면 잔고 감소)`);

  const optState = [
    `대차(유통)≥${CFG.loanFloatPctMin}%`,
    `연속상환≥${CFG.contRepayMin}일`,
    `공매도거래대금≥${CFG.shortAmtEokMin}억 ${CFG.enableShortAmt ? "ON" : "OFF"}`,
    `3일 대차증감≤${CFG.recent3NetRepayMax.toLocaleString()}주 ${CFG.enableRecent3Repay ? "ON" : "OFF"}`,
    `5일평균 거래대금≥${CFG.turnover5dAvgEokMin}억 ${CFG.enableTurnover5d ? "ON" : "OFF"}`,
  ];
  dv.paragraph(`- 활성 필터: ${optState.join(" | ")}`);

  // ── ALL PASS
  dv.header(2, "✅ 5필터 동시충족 (ALL PASS)");
  dv.paragraph(`- 통과: **${allPass.length}** / 전체: **${rows.length}**`);
  if (allPass.length) {
    const headers = ["종목", "대차(유통)", "대차(총)", "연속상환"];
    if (CFG.enableShortAmt) headers.push("공매도거래대금(억)");
    if (CFG.enableRecent3Repay) headers.push("3일 대차증감(주)");
    if (CFG.enableTurnover5d) headers.push("5일평균 거래대금(억)");

    dv.table(headers, allPass.map(r => {
      const row = [
        r.link,
        fmtPct(r.loanPctFloat),
        fmtPct(r.loanPctTotal),
        (r.contRepay === null ? "-" : `${r.contRepay}일`),
      ];
      if (CFG.enableShortAmt) row.push(fmtEok(r.shortAmtEok));
      if (CFG.enableRecent3Repay) row.push(r.loanDelta3 === null ? "-" : r.loanDelta3.toLocaleString());
      if (CFG.enableTurnover5d) row.push(fmtEok(r.turnover5dAvgEok));
      return row;
    }));
  } else {
    dv.list(["(조건 통과 없음)"]);
  }

  dv.el("hr","");

  // ── 숏커버 BEST (칼럼 축소 + 옵션컬럼만 노출)
  dv.header(2, `🏆 숏커버 BEST ${CFG.bestShortN} (점수 기반)`);
  {
    const headers = ["종목","점수","대차(유통)","대차(총)","연속상환"];
    if (CFG.enableShortAmt) headers.push("공매도거래대금(억)");
    if (CFG.enableRecent3Repay) headers.push("3일 대차증감(주)");
    if (CFG.enableTurnover5d) headers.push("5일평균 거래대금(억)");

    dv.table(headers, bestShort.map(r => {
      const row = [
        r.link,
        r.score.toFixed(1),
        fmtPct(r.loanPctFloat),
        fmtPct(r.loanPctTotal),
        (r.contRepay === null ? "-" : `${r.contRepay}일`),
      ];
      if (CFG.enableShortAmt) row.push(fmtEok(r.shortAmtEok));
      if (CFG.enableRecent3Repay) row.push(r.loanDelta3 === null ? "-" : r.loanDelta3.toLocaleString());
      if (CFG.enableTurnover5d) row.push(fmtEok(r.turnover5dAvgEok));
      return row;
    }));
  }

  dv.el("hr","");

  // ── 신용 BEST (칼럼 축소)
  dv.header(2, `💳 신용 BEST ${CFG.bestCreditN} (점수 기반)`);
  dv.table(
    ["종목","점수","신용(유통)","신용공여율(%)","3일신용증감(주)","1일신용증감(주)"],
    bestCredit.map(r => [
      r.link,
      r.score.toFixed(1),
      fmtPct(r.creditPctFloat),
      (r.creditSupplyPct === null ? "-" : r.creditSupplyPct.toFixed(2)),
      (r.creditDelta3 === null ? "-" : r.creditDelta3.toLocaleString()),
      (r.creditRepayLatest === null ? "-" : r.creditRepayLatest.toLocaleString()),
    ])
  );

  if (CFG.DEBUG) {
    dv.el("hr","");
    dv.header(3, "DEBUG");
    dv.paragraph(`- 파싱/로드 실패: **${errors.length}**`);
    if (errors.length) {
      const msg = errors.slice(0, 30).map(e => `> - ${e.link} (${e.where}) : ${e.msg}`).join("\n");
      dv.paragraph(`> [!warning]- 파싱 실패 노트 (최대 30개)\n${msg}`);
    }
    const sample = rows.slice(0, 10).map(r => [
      r.link,
      r.loanBalLatest === null ? "loan=null" : `loan=${r.loanBalLatest.toLocaleString()}`,
      r.loanPctFloat === null ? "loanF=null" : `loanF=${r.loanPctFloat.toFixed(2)}%`,
      r.creditBalLatest === null ? "cred=null" : `cred=${r.creditBalLatest.toLocaleString()}`,
      r.creditDelta3 === null ? "c3=null" : `c3=${r.creditDelta3.toLocaleString()}`,
    ]);
    dv.table(["종목","대차잔고","대차(유통)","신용잔고","3일 신용증감"], sample);
  }
